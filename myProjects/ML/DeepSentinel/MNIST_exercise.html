def visualize_y_distribution(y, save_name=None):
    
    '''
    Plot distribution of y label in a given list of data
    '''
    
    x_ticks, counts = np.unique(y, return_counts=True)
    
    plt.title('Data distribution over y labels')
    plt.plot(counts)
    plt.xticks(x_ticks)
    plt.ylim(ymin=0)
    plt.show()
    
    if save_name:
        plt.savefig(save_name)

def visualize_prediction(predict, save_name=None):
    
    '''
    Plot the distribution of predition classes
    '''

    # visualize prediction in each classes
    x_ticks, counts = np.unique(predict, return_counts=True)

    counts = [counts[np.where(x_ticks==i)[0]] if i in x_ticks else 0 for i in range(10)]
    x_ticks = np.arange(10)

    plt.title('Prediction in each classes')
    plt.plot(counts)
    plt.xticks(x_ticks)
    plt.ylim(ymin=0)
    plt.show()
    
    if save_name:
        plt.savefig(save_name)
    
    return counts


def plot_sample(sample, show=True):
    
    '''
    Plot a sample

    plt.imshow(sample, cmap='gray')

def plot_samples_overlay(dataset, show=True):
    
    '''
    Plot a set of samples' overlay
    '''
    
    dataset = dataset.astype('int')
    avg = np.round(sum(dataset) / dataset.shape[0])
    plot_sample(avg, show=show)


def plot_overlay_compare(origin, treatment):
    
    '''
    Plot two overrlay side-by-side for comparison
    '''
    
    fig = plt.figure()
    
    fig1 = fig.add_subplot(1,2,1)
    fig1.set_title('origin')
    plot_samples_overlay(origin, show=False)
    
    fig2 = fig.add_subplot(1,2,2)
    plot_samples_overlay(treatment, show=False)
    fig2.set_title('treatment')
    
    plt.show()
    
    
def plot_overlay_of_classes(X, y, classes=None, name=None):
    
    '''
    Designate classes in list to visualize at once
    '''

    fig = plt.figure()
    
    if type(classes)==list:
        iterate = classes
    else:
        iterate = range(10)
    
    for i,v in enumerate(iterate):
        sub = fig.add_subplot(1,len(iterate),i+1)
        sub.axis('off')
        sub.set_title(str(v))
        plot_samples_overlay(X[y==v], show=False)
    
    fig.suptitle(name if name else 'No_title')
    plt.show()

retain_ratio = 0.01
victims = [2,3,6]

def rm_99_percent_data_of(X, y, victims):
    
    retain_idxes = []
    victim_idxes = [[] for i in range(len(victims))] # note: [[]]*len(victims) will copy reference instead of deep copy
    for i,yy in enumerate(y):
        is_victim = False
        for j,vic in enumerate(victims):
            if yy==vic:
                victim_idxes[j].append(i)
                is_victim = True
                break
        if not is_victim:
            retain_idxes.append(i)
    
    # random retain 1% data
    removed_idxes = []
    for i in range(len(victim_idxes)):
        retain_size = round(len(victim_idxes[i])*retain_ratio)
        np.random.shuffle(victim_idxes[i])
        retain_idxes += victim_idxes[i][:retain_size]
        removed_idxes += victim_idxes[i][retain_size:]

    return X[retain_idxes], y[retain_idxes], removed_idxes

train_ratio = 0.8 # 0.7 is also good, but I prefer 0.8

def expand_y(y):
    
    '''
    Reshape the y label for later training
    '''
    
    ret = np.zeros( (y.shape[0], 10) )
    for i,yy in enumerate(y):
        ret[i,yy] = 1
    return ret


def prepare_train_val(X, y):
    
    '''
    Prepare data.
    Include split train/val set, and reshape y to one-hot form.
    '''

    train_size = round(X.shape[0]*train_ratio)

    shuffle_idxes = np.arange(X.shape[0])
    np.random.shuffle(shuffle_idxes)

    train_idxes = shuffle_idxes[:train_size]
    val_idxes = shuffle_idxes[train_size:]

    X_val = X[val_idxes]
    X_train = X[train_idxes]
    y_val = expand_y(y[val_idxes])
    y_train = expand_y(y[train_idxes])
    
    return X_train, y_train, X_val, y_val

from keras.models import Model
from keras.layers import Dense, Input, Reshape, Activation, Flatten, Dropout, BatchNormalization
from keras.layers.convolutional import Conv2D, MaxPooling2D
from keras.optimizers import Nadam
from keras.callbacks import EarlyStopping, ModelCheckpoint

def build_model():
    input_shape = X_train.shape[1:]
    img_input = Input(shape=input_shape)
    data = Reshape(target_shape=(1,input_shape[0],input_shape[1]))(img_input)
    
    data = Conv2D(filters=32, kernel_size=2, padding='same')(data)
    data = MaxPooling2D(pool_size=(2,2), padding='same')(data)
    data = Conv2D(filters=32, kernel_size=2, padding='same')(data)
    data = MaxPooling2D(pool_size=(2,2), padding='same')(data)
    
    data = Activation('relu')(data)
    
    data = Conv2D(filters=64, kernel_size=2, padding='same')(data)
    data = MaxPooling2D(pool_size=(2,2), padding='same')(data)
    data = Conv2D(filters=64, kernel_size=2, padding='same')(data)
    data = MaxPooling2D(pool_size=(2,2), padding='same')(data)
    
    data = Activation('relu')(data)
    
    data = Conv2D(filters=128, kernel_size=2, padding='same')(data)
    data = MaxPooling2D(pool_size=(2,2), padding='same')(data)
    data = Conv2D(filters=128, kernel_size=2, padding='same')(data)
    data = MaxPooling2D(pool_size=(2,2), padding='same')(data)
    
    data = Flatten()(data) # shape here is (1,1,128)
    
    data = Dense(64)(data)
    data = Dense(64)(data)
    data = Dense(32)(data)
    data = Dense(32)(data)
    
    out = Dense(10, activation='softmax')(data)

    model = Model(inputs=img_input, outputs=out)
    
    model.compile(loss='categorical_crossentropy',
            optimizer=Nadam(lr=0.002, clipnorm=0.5),
            metrics=['acc'])
    
    return model

def train(model, train, validation, save_path=None, patience=3, batch_size=512,monitor_policy = 'val_acc'):
    
    save_path = save_path if save_path else './tmp.model'
    
    X_train, y_train = train
    X_val, y_val = validation

    early_stopping =EarlyStopping(monitor=monitor_policy, patience=patience)
    model_checkpoint = ModelCheckpoint(save_path, save_best_only=True, save_weights_only=True)

    try:
        # model.summary()
        hist = model.fit(X_train, 
                         y_train,
                         validation_data=(X_val, y_val),
                         epochs=200, 
                         batch_size=batch_size, 
                         shuffle=True,
                         callbacks=[early_stopping, model_checkpoint], 
                         verbose=1
                        )
    except Exception as e:
        if type(e)==KeyboardInterrupt:
            print('Interrupt by user')
        else:
            raise(e)

    model.load_weights(save_path)
    bst_val_score = max(hist.history['val_acc'])
    return bst_val_score

def transform_prediction(predict):
    
    '''
    Transform prediction from one-hot to numeric
    '''
    
    return np.array([np.argmax(pp) for pp in predict])


def calc_acc(predict, truth):
    
    '''
    Compute accuracy
    '''
    
    predict_trans = transform_prediction(predict)
    return sum(predict_trans==truth)/predict.shape[0], predict_trans

Fast conclusion¶

  1. Test set has -0.12 (0.985->0.86) performance difference in terms of accuracy, comparing to validation accuracy.

     Accuracy drop is totally as expected. 
 This is totally data problem. 
 Can either be explained by:

     A. The distribution of training data is different to real world data causes regression.
        The samples in classes [2,3,6] are all too few. Making our model prefers to give other classes higher probability. 

     B. Or sometimes even worse, model will totally give up to learn the classes whom rarely appears.
        Since those classes rarely update weights, the previous hidden layers are all fully controlled by other classes.
        Thus the model can't really learn any information to support the prediction on such rare classes.

 This issue shouldn't happen at all if we "carefully" obtain the training data from true user events.
 But engineers still need to clean up the mess if such tragedy happens :(

  2. [2,3,6] classes has lower trigger rate compare to other classes is expected.

     But it surprised me that the model still outputs so many predictions in [2,3,6] classes. 
 I thought it will totally ignore those 3 classes.
 Maybe my model is too complex such that it overfits to the samples in classes [2,3,6], so it can identify very similar samples in the testing set.
 It is an interesting result, but I got to be hurry ;(

  3. The class "5" or "8" has very high trigger rate.

     Maybe its because 5 and 8 is very similar to either 2, 3 or 6 in hand-writing.
 We can visualize it via some well-known libraries if I have time.

Q2: How to solve it?¶

I came up with 3 solutions:

  1. [Naive] Simply upsampling or tuning the class weights during training.

     Pros:
     1. Easy to do.
     2. Very possible to work.
 Cons:
     1. [2,3,6] classes are extremely imbalanced, we might overfit to the very small training data of classes [2,3,6].
     2. The improvement (accuracy classes [2,3,6] or overall accuracy) is not gauranteed, although this often works.

 So I won't do this.
 Although it can be used as a baseline, but I will use data augmentation later, making this baseline meaningless.

  2. [Acceptable] Combine data augmentation and up/down sampling.

     Basically, the only difference comparing to previous solution is adding data augmentation. 
 Since the key problem of identifying classes [2,3,6] is they are lack of data. So data augmentation should help them a lot. 
 And we can tune the appearance probability of each class, since we already "somehow" knows that number 0~9 appears almost equally in real world data.

  3. [Ultimate weapon] GAN (will not and no need do it here)

     Although data augmentation should work, but we still aren't very sure if the data of classes [2,3,6] is enough or not. 
 If the answer is no, we probably need to use GAN to generate new and high quality data.
 There are still plenty of problems if we adapt this approach:

 Q1: We already lack of data, can the GAN training success?

         I don't know, but worth a try if necessary.

 Q2: How to gaurantee "high-quality"?

         Yep, another challenge. Maybe use an autoencoder to do data selection is applicable? I'm not very sure about that.
         Apple just published a paper about how to generate high-quality image data in CVPR, maybe we can find some idea there. 
         (I still don't have time to read it ;( )

  4. What's next?

     I don't really know. 
 But I'm very sure the golden key is generating more high quality data to solve the data problem. 
 Maybe try some more traditional ways to generate more data. 
 For example, find a way to interpolate a new "6" from two different "6".

def duplicate_class_samples(class_label, X, y, target_size):
    class_position = y==class_label
    X_class = X[class_position]
    y_class = y[class_position]
    diff = target_size - X_class.shape[0]
    dup_index = np.random.randint(0,X_class.shape[0],size=diff)
    return X_class[dup_index], y_class[dup_index]

Note:

I only augment classes [2,3,6].

# Augmentation tool:   imgaug
# Github :             https://github.com/aleju/imgaug
# Install :            pip install git+https://github.com/aleju/imgaug

import imgaug as ia
from imgaug import augmenters as iaa

ia.seed(1)
'''
Input:
    src: augmentation source images
    label: class label
    size: augment output size

Output:
    X,y augmentated new samples (not including input)
'''
def augmentation(src, label, size):

    seq = iaa.Sequential([
#         iaa.PerspectiveTransform(scale=0.05), # can't use it in greyscale QQ
        iaa.Affine(
            translate_percent={"x": (-0.04, 0.04), "y": (-0.04, 0.04)}, # 0.04 * 28 (px) = 1.12px ~= 1px
            rotate=(-3, 3),
            shear=(-3, 3)
            
        )
    ], random_order=True) # apply augmenters in random order
    
    augmented = np.array([seq.augment_image(src[np.random.randint(0,src.shape[0])]) for i in range(size)])
    return augmented, np.repeat(label, size)

Conclution¶

  1. The result already looks pretty good in terms of both accuracy prediction distribution.

     Accuracy difference between validation set and testing set is only 0.027 ( 0.99 -> 0.963 ).
 Seems like there's no need to use GAN or other fancy technologies.

 By fine-tuning the augmentatio hyper-parameters, I optimistically guess we can enclose the accuracy gap to 0.02.

  2. Why data augmentation works so good?

     Since hand-writing digits data is very simple and easy to distinguish.
 I think the key problems are:

     A. Label distribution different (solve by up-sampling)

     B. Classes [2,3,6] appears too rarely. (solve by up-sampling)
        Model can't properly store information into hidden nodes since other classes update too frequetly.

     C. Sample size of classes [2,3,6] is too small, model overfits to it. (solve by data augmentation)
        The overfitting is happened in the dense layers (or fully-connected layers) since dense layer is positional sensitive.

 Since the key problems are all solved, so good performance is expected. 
 But since we still lack of data diversity in classes [2,3,6]. 
 I wouldn't be surprised if there's remaining regression in those classes.

  3. There are some things I didn't do since the key problem of this exercise is data distribution.

     A. K-fold validation.

     The accuracy varies for different random seed.
     (But I'm pretty sure that the result is consistently augmetation>duplication>skew).
     Also, the hyper-paramters / model structure should be tuned via validation.

 B. Other generalization components in model.

     Although we can solve overfitting problem by data augmentation, 
     theoretically we can also solve it by adding dropout layer to our model's dense layer. 
     But I found it doesn't work, maybe the sample size of classes [2,3,6] is really too small.
     Or, maybe we need more hyper-parameter tuning, not very sure about that.
